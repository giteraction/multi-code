from collections import deque
def solution(progresses, speeds):

    answer = []
    n=0
    queue=deque()
    leng=len(speeds)
    for i in range(leng):
        queue.append((100-progresses[i])/speeds[i])
    for i in range(leng):
        if queue[i]<queue[i+1]:
            queue.popleft()
            n+=1
        elif queue[i]>queue[i+1]:
            answer.append(n)
            n=0
        
    return answer
