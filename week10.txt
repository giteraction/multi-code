# 직사각형 길이 찾기

t=int(input())
for i in range(t):
    a,b,c=map(int,input().split())
    d=0
    if a==b:
        d=c
    elif b==c:
        d=a
    else:
        d=b
    print('#{}'.format(i+1) ,d)
