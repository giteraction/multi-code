#두가지 빵의 딜레마

t= int(input())

for _ in range(t):
    n=0
    a,b,c=map(int,input().split())
    n+=c//min(a,b)
    if c%min(a,b)>max(a,b):
        n+=c//max(a,b)
    print('#{}'.format(_+1) ,n)

#문자열 거울상

t= int(input())

for _ in range(t):
    word=list(input())
    word.reverse()
    for i in range(len(word)):
        if word[i]=='b':
            word[i]='d'
        elif word[i]=='d':
            word[i]='b'            
        elif word[i]=='p':
            word[i]='q'    
        elif word[i]=='q':
            word[i]='p'    
    print('#{}'.format(_+1),end=' ')
    for i in range(len(word)):
        print(word[i],end='')
